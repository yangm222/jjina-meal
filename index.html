const START_DATE = new Date(2025, 4, 9); // 5/9 이유식 시작

const TODAY = new Date(2025, 4, 14); // 오늘 5/14


const DOW = ["월","화","수","목","금","토","일"];

const DOW_EN = ["Mon","Tue","Wed","Thu","Fri","Sat","Sun"];


function dateKey(date) {

  return `${date.getMonth()+1}/${date.getDate()}`;

}

function getDDay(date) {

  const diff = Math.floor((date - START_DATE) / 86400000);

  return `D+${diff + 1}`;

}

function getDow(date) {

  const d = date.getDay();

  return DOW[d === 0 ? 6 : d - 1];

}


function badgeClass(b) {

  const map = {"NEW":"b-new","직접":"b-직접","시판":"b-시판","시판+직접":"b-혼합","주의":"b-warn","반복":"b-rep","키트":"b-kit"};

  return map[b] || "b-직접";

}

let currentMonth = TODAY.getMonth(); // 0-indexed

let selectedDate = new Date(TODAY);


function getWeekDates(date) {

  const day = date.getDay();

  const mon = new Date(date);

  mon.setDate(date.getDate() - (day === 0 ? 6 : day - 1));

  const days = [];

  for (let i = 0; i < 7; i++) {

    const d = new Date(mon);

    d.setDate(mon.getDate() + i);

    days.push(d);

  }

  return days;

}


function renderWeekBar() {

  const bar = document.getElementById('week-bar');

  const ml = document.getElementById('month-label');

  ml.textContent = `${currentMonth+1}월`;

  

  const weekDays = getWeekDates(selectedDate);

  bar.innerHTML = weekDays.map(d => {

    const key = dateKey(d);

    const data = mealData[key];

    const isToday = d.toDateString() === TODAY.toDateString();

    const isSelected = d.toDateString() === selectedDate.toDateString();

    const hasDone = data && data.done;

    const hasData = !!data;

    

    let cls = 'week-day';

    if (isToday) cls += ' today';

    if (isSelected) cls += ' selected';

    if (hasDone) cls += ' done';

    else if (hasData) cls += ' has-data';

    

    return `<div class="${cls}" onclick="selectDate(new Date(${d.getFullYear()},${d.getMonth()},${d.getDate()}))">

      <div class="day-num">${d.getDate()}</div>

      <div class="day-dot"></div>

    </div>`;

  }).join('');

}


function selectDate(date) {

  selectedDate = date;

  currentMonth = date.getMonth();

  renderWeekBar();

  renderDayContent();

}


function changeMonth(dir) {

  currentMonth += dir;

  if (currentMonth < 0) currentMonth = 0;

  if (currentMonth > 11) currentMonth = 11;

  // 해당 월의 첫 날로 이동

  const newDate = new Date(2025, currentMonth, 1);

  // 이유식 데이터 있는 첫 날 찾기

  for (let i = 1; i <= 31; i++) {

    const d = new Date(2025, currentMonth, i);

    if (d.getMonth() !== currentMonth) break;

    if (mealData[dateKey(d)]) { selectedDate = d; break; }

  }

  renderWeekBar();

  renderDayContent();

}


function getNewIngredientsThisWeek(date) {

  const weekDays = getWeekDates(date);

  const newItems = [];

  weekDays.forEach(d => {

    const key = dateKey(d);

    const data = mealData[key];

    if (data && data.badges && data.badges.includes('NEW')) {

      data.items.forEach(item => {

        if (!newItems.includes(item.name)) newItems.push(item.name);

      });

    }

  });

  return newItems;

}
